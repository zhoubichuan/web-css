# 三.缩放

::: demo

```html
<style>
.scale-ele {
 height:200px;
 width:200px;
}
</style>
  <div id="scale-ele" class="scale-ele" >
    内容
    <div class="wrapper">
      <button class="children2" onclick="add(0.5,0, 0)">放大</button>
      <button class="children2" onclick="reduce( 0.5,0, 0)">缩小</button>
    </div>
  </div>
<script>
  let scale = 1;
  function add(num, originX, originY) {
    let ele = document.getElementById("scale-ele")
    scale += num
    ele.style.transformOrigin = `${originX}% ${originY}%`;
    ele.style.transform = `scale(${scale})`;
  }
  function reduce(num, originX, originY) {
    let ele = document.getElementById("scale-ele")
    scale -= num
    ele.style.transformOrigin = `${originX}% ${originY}%`;
    ele.style.transform = `scale(${scale})`;
  }
</script>
```

:::
