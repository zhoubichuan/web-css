# 四.换肤

::: demo

```html
<style>
.spring-theme { --bg: #171616; --fontColor: green; }
.summer-theme { --bg: #f5f5f5; --fontColor: blue; }
.autumn-theme { --bg: #171616; --fontColor: yellow; }
.winter-theme { --bg: #f5f5f5; --fontColor: gray; }
.wrapper {
  backgroundColor: var(--bg);
  color: var(--fontColor)
}
</style>
<div id="box" class="spring-theme">
  <button onclick="changeTheme('spring')">春天主题</button>
  <button onclick="changeTheme('summer')">夏天主题</button>
  <button onclick="changeTheme('autumn')">秋天主题</button>
  <button onclick="changeTheme('winter')">冬天主题</button>
  <div class="wrapper">
    <div class="children2">123</div>
  </div>
</div>
<script>
  function changeTheme(theme) {
    let ele = document.getElementById("box")
    if(theme){
      const themeName = theme+"-theme"
      ele.className = themeName
    }else{
      const themeName = localStorage.getItem("theme")
      ele.className =themeName|| "spring-theme"
    }
    localStorage.setItem("theme",ele.className)
  }
  changeTheme()
</script>
```

:::
